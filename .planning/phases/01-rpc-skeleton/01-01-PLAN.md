    1. ui-ink/src/tools/fs.ts 신규 작성 — D-14/D-15 schema 그대로 + Python 의 줄 번호 prefix 동등:

       ```typescript
       // RPC-03 (Phase 1): read_file — Python tools/fs.py:read_file 와 동등 schema (D-14, D-15).
       // alias 처리는 agent.py 측 (D-16) — TS 는 path 만 받음.
       import {promises as fs} from 'node:fs'
       import type {ToolResult} from './registry.js'

       export async function readFile(rawArgs: Record&lt;string, unknown&gt;): Promise&lt;ToolResult&gt; {
         const path = typeof rawArgs['path'] === 'string' ? rawArgs['path'] : ''
         const offset = typeof rawArgs['offset'] === 'number' ? rawArgs['offset'] : 1
         const limit = typeof rawArgs['limit'] === 'number' ? rawArgs['limit'] : 0
         if (!path) {
           return {ok: false, error: 'path 누락'}
         }
         try {
           const buf = await fs.readFile(path, 'utf-8')
           // keepends 동등 — Python f.readlines() 가 줄바꿈 포함
           const lines = buf.length === 0 ? [] : buf.split(/(?&lt;=\n)/)
           const total = lines.length
           const start = Math.max(1, offset) - 1
           const end = limit &gt; 0 ? Math.min(start + limit, total) : total
           const sliced = start &gt;= total ? [] : lines.slice(start, end)
           // cat -n 스타일 — Python 의 f'{n:4d}\t{l}' 동등 (4자리 right-aligned)
           const content = sliced
             .map((l, i) =&gt; `${String(start + i + 1).padStart(4, ' ')}\t${l}`)
             .join('')
           return {
             ok: true,
             content,
             total_lines: total,
             start_line: start + 1,
             end_line: start + sliced.length,
           }
         } catch (e) {
           const m = e instanceof Error ? e.message : String(e)
           return {ok: false, error: m}
         }
       }
       ```

    2. ui-ink/src/tools/registry.ts 신규 작성 — store/confirm.ts:9-13 의 boundClient 패턴 + components/tools/index.ts 의 dict-lookup 합성:

       ```typescript
       // RPC-02/03: 클라 측 도구 실행 registry. dispatch.ts 의 'tool_request' 가 진입점.
       // 새 도구 추가 절차: tools/&lt;file&gt;.ts 작성 → TOOL_RUNTIME 에 한 줄 추가 → 끝.
       import {readFile} from './fs.js'
       import type {HarnessClient} from '../ws/client.js'
       import type {ClientMsg} from '../protocol.js'

       export type ToolResult =
         | ({ok: true} &amp; Record&lt;string, unknown&gt;)
         | {ok: false; error: string}

       export type ToolFn = (args: Record&lt;string, unknown&gt;) =&gt; Promise&lt;ToolResult&gt;

       // Phase 1 = read_file 만. Phase 2 에서 write/edit/list/grep 추가.
       export const TOOL_RUNTIME: Record&lt;string, ToolFn&gt; = {
         read_file: readFile,
       }

       // store/confirm.ts:9-13 의 boundClient 패턴 — 순환 의존 회피 (registry → client → dispatch → registry 차단)
       let boundClient: HarnessClient | null = null
       export function bindToolClient(client: HarnessClient | null): void {
         boundClient = client
       }

       export async function runClientTool(
         call_id: string,
         name: string,
         args: Record&lt;string, unknown&gt;,
       ): Promise&lt;void&gt; {
         const fn = TOOL_RUNTIME[name]
         let response: ClientMsg
         if (!fn) {
           response = {
             type: 'tool_result',
             call_id,
             ok: false,
             error: {kind: 'unknown_tool', message: `클라 측 도구 미등록: ${name}`},
           }
         } else {
           try {
             const r = await fn(args)
             if (r.ok) {
               // ok 플래그 자체는 envelope 의 ok 로 들어가고, 나머지 필드만 result 에 포장
               const {ok: _ok, ...rest} = r
               response = {type: 'tool_result', call_id, ok: true, result: rest}
             } else {
               response = {
                 type: 'tool_result',
                 call_id,
                 ok: false,
                 error: {kind: 'tool_error', message: r.error},
               }
             }
           } catch (e) {
             const m = e instanceof Error ? e.message : String(e)
             response = {
               type: 'tool_result',
               call_id,
               ok: false,
               error: {kind: 'exception', message: m},
             }
           }
         }
         boundClient?.send(response)
       }
       ```
