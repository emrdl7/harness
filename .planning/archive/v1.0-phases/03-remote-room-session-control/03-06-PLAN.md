```typescript
import {structuredPatch} from 'diff'

export const DiffPreview: React.FC<DiffPreviewProps> = ({path, newContent, oldContent}) => {
  if (!oldContent) {
    return (
      <Box flexDirection='column'>
        <Text color='green'>+ (신규 파일) {path}</Text>
      </Box>
    )
  }

  const patch = structuredPatch(path, path, oldContent, newContent)
  const lines: Array<{text: string; color: string | undefined}> = []
  for (const hunk of patch.hunks) {
    lines.push({text: `@@ -${hunk.oldStart},${hunk.oldLines} +${hunk.newStart},${hunk.newLines} @@`, color: 'cyan'})
    for (const line of hunk.lines) {
      const color = line.startsWith('+') ? 'green' : line.startsWith('-') ? 'red' : undefined
      lines.push({text: line, color})
    }
  }

  return (
    <Box flexDirection='column'>
      {lines.map((l, i) => (
        <Text key={`diff-${String(i)}`} color={l.color}>{l.text}</Text>
      ))}
    </Box>
  )
}
```
   주의: `diff@9`은 Phase 1에서 이미 설치됨 (RESEARCH.md Don't Hand-Roll 확인). `structuredPatch`가 named export인지 확인 후 import 조정.
